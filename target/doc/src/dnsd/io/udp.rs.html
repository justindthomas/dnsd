1//! UDP/53 listener over `vcl-rs`.
2//!
3//! One `VclDgramSocket` per listener. Every recv → ACL → dispatch →
4//! send is handled on the caller's Tokio task; heavy lifting happens
5//! inside the handler (which is expected to spawn its own tasks for
6//! upstream queries). Datagrams up to 4096 B are accepted — enough
7//! for an EDNS0-advertised MTU that fits in our normal MTU, larger
8//! responses drop TC=1 and the client retries over TCP.
9
10use std::sync::atomic::Ordering;
11use std::sync::Arc;
12
13use anyhow::{Context, Result};
14use vcl_rs::{VclDgramSocket, VclReactor};
15
16use crate::acl::ClientAcl;
17use crate::config::Listener;
18use crate::handler::SharedHandler;
19use crate::metrics::Metrics;
20
21const UDP_BUF_SIZE: usize = 4096;
22
23pub struct UdpListener;
24
25impl UdpListener {
26    /// Bind the listener via VCL and spawn the serve loop on the
27    /// current Tokio runtime. Returns once the socket is bound; the
28    /// loop runs until `reactor` / `handler` are dropped.
29    pub async fn spawn(
30        listener_cfg: Listener,
31        reactor: VclReactor,
32        handler: SharedHandler,
33        metrics: Arc<Metrics>,
34    ) -> Result<tokio::task::JoinHandle<()>> {
35        let bind = std::net::SocketAddr::new(listener_cfg.address, listener_cfg.port);
36        let sock = VclDgramSocket::bind(bind, reactor)
37            .with_context(|| format!("UDP bind {bind}"))?;
38        let acl = ClientAcl::new(listener_cfg.allow_from.clone());
39        let name = listener_cfg.name.clone();
40        tracing::info!(listener = %name, addr = %bind, "UDP listener up");
41
42        let handle = tokio::spawn(async move {
43            serve_loop(sock, acl, handler, metrics, name).await;
44        });
45        Ok(handle)
46    }
47}
48
49async fn serve_loop(
50    sock: VclDgramSocket,
51    acl: ClientAcl,
52    handler: SharedHandler,
53    metrics: Arc<Metrics>,
54    name: String,
55) {
56    let sock = Arc::new(sock);
57    let mut buf = vec![0u8; UDP_BUF_SIZE];
58    loop {
59        let (n, peer) = match sock.recv_from(&mut buf).await {
60            Ok(v) => v,
61            Err(e) => {
62                tracing::error!(listener = %name, "recv_from: {e}");
63                // Transient VCL errors shouldn't kill the loop.
64                tokio::time::sleep(std::time::Duration::from_millis(10)).await;
65                continue;
66            }
67        };
68
69        if !acl.allows(peer.ip()) {
70            metrics.acl_denied.fetch_add(1, Ordering::Relaxed);
71            tracing::debug!(listener = %name, %peer, "ACL denied");
72            continue;
73        }
74
75        metrics.queries_udp.fetch_add(1, Ordering::Relaxed);
76
77        // Offload to a spawned task so slow handlers don't back up
78        // the receive loop. `sock` is cheap to clone through Arc.
79        let handler = handler.clone();
80        let sock = sock.clone();
81        let name = name.clone();
82        let query = buf[..n].to_vec();
83        tokio::spawn(async move {
84            if let Some(response) = handler.handle_bytes(&query, peer.ip()).await {
85                if let Err(e) = sock.send_to(&response, peer).await {
86                    tracing::debug!(listener = %name, %peer, "send_to: {e}");
87                }
88            }
89        });
90    }
91}
