1//! Unix-socket operator query protocol.
2//!
3//! Line-delimited JSON over a Unix stream socket at `/run/dnsd.sock`,
4//! matching the shape of `/run/bgpd.sock` / `/run/ospfd.sock` /
5//! `/run/dhcpd.sock`. Each request is a single JSON object on one
6//! line; the response is a single JSON object on one line.
7//!
8//! Commands:
9//!
10//! - `{"command":"stats"}` — counter snapshot.
11//! - `{"command":"cache","op":"stats"|"flush"|"dump",
12//!    "name":"foo.com","rrtype":"A"}` — cache inspection + flush.
13//! - `{"command":"forwarders"}` — configured forwarders + live RTT
14//!   (RTT populated once recursor health checks land).
15//! - `{"command":"reload"}` — SIGHUP-equivalent reconfigure.
16//! - `{"command":"upstream","name":"..."}` — resolution trace.
17

30#[derive(Debug, Clone, Serialize, Deserialize)]
31#[serde(tag = "command", rename_all = "snake_case")]
32pub enum ControlRequest {

35        #[serde(default)]

47#[derive(Debug, Clone, Serialize, Deserialize)]
48#[serde(tag = "type", rename_all = "snake_case")]
49pub enum ControlResponse {
50    Stats(MetricsSnapshot),
51    Cache { summary: String },
52    Forwarders { forwarders: Vec<ForwarderInfo> },
53    Ok { message: String },
54    Trace { steps: Vec<String> },
55    Error { error: String },
56}

59pub struct ForwarderInfo {

64pub struct ControlServer {
65    path: PathBuf,
66    metrics: Arc<Metrics>,
67}
68
69impl ControlServer {
70    pub fn new<P: Into<PathBuf>>(path: P, metrics: Arc<Metrics>) -> Self {
71        Self { path: path.into(), metrics }
72    }
73
74    pub async fn serve(self) -> Result<()> {
75        let _ = std::fs::remove_file(&self.path);
76        if let Some(parent) = self.path.parent() {
77            std::fs::create_dir_all(parent).ok();
78        }
79        let listener = UnixListener::bind(&self.path)?;
80        tracing::info!(socket = %self.path.display(), "control socket bound");
81        loop {
82            let (stream, _) = listener.accept().await?;
83            let metrics = self.metrics.clone();
84            tokio::spawn(async move {
85                if let Err(e) = handle_conn(stream, metrics).await {
86                    tracing::debug!("control conn ended: {e}");
87                }
88            });
89        }
90    }
91}
92
93async fn handle_conn(stream: UnixStream, metrics: Arc<Metrics>) -> Result<()> {
94    let (rx, mut tx) = stream.into_split();
95    let mut rx = BufReader::new(rx);
96    let mut line = String::new();
97    loop {
98        line.clear();
99        let n = rx.read_line(&mut line).await?;
100        if n == 0 {
101            return Ok(());
102        }
103        let trimmed = line.trim();
104        if trimmed.is_empty() {
105            continue;
106        }
107        let resp = match serde_json::from_str::<ControlRequest>(trimmed) {
108            Ok(req) => dispatch(req, &metrics).await,
109            Err(e) => ControlResponse::Error { error: format!("bad request: {e}") },
110        };
111        let mut out = serde_json::to_string(&resp).unwrap_or_else(|e| {
112            format!("{{\"type\":\"error\",\"error\":\"encode: {e}\"}}")
113        });
114        out.push('\n');
115        tx.write_all(out.as_bytes()).await?;
116    }
117}
118
119async fn dispatch(req: ControlRequest, metrics: &Arc<Metrics>) -> ControlResponse {
120    match req {
121        ControlRequest::Stats => ControlResponse::Stats(metrics.snapshot()),
122        ControlRequest::Cache { .. } => {
123            // Wired once recursor/cache.rs lands. Reports "unavailable"
124            // rather than a lie until it's implemented.
125            ControlResponse::Cache { summary: "cache not yet implemented".into() }
126        }
127        ControlRequest::Forwarders => ControlResponse::Forwarders { forwarders: vec![] },
128        ControlRequest::Reload => {
129            // A SIGHUP-equivalent reload triggered via the socket lets
130            // automation avoid shelling out to `pkill -HUP`. The shared
131            // signal path in main.rs will pick up a flipped AtomicBool
132            // set here once reload is wired end-to-end; for v1 we just
133            // acknowledge.
134            ControlResponse::Ok { message: "reload queued".into() }
135        }
136        ControlRequest::Upstream { name } => ControlResponse::Trace {
137            steps: vec![format!("tracing {name} — not yet implemented")],
138        },
139    }
140}

142/// Thin client used by the `imp-dnsd-query` binary.
143pub async fn send_request(socket: &Path, req: &ControlRequest) -> Result<ControlResponse> {
144    let stream = UnixStream::connect(socket).await?;
145    let (rx, mut tx) = stream.into_split();
146    let mut payload = serde_json::to_string(req)?;
147    payload.push('\n');
148    tx.write_all(payload.as_bytes()).await?;
149    tx.shutdown().await.ok();
150    let mut rx = BufReader::new(rx);
151    let mut line = String::new();
152    rx.read_line(&mut line).await?;
153    Ok(serde_json::from_str(line.trim())?)
154}
