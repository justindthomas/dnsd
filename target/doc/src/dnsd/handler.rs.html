1//! The `DnsHandler` trait is the boundary between the transport layer
2//! (UDP / TCP / DoT / DoH) and the query-processing layer (cache,
3//! forwarder, recursor, DNS64, DNSSEC).
4//!
5//! Transports parse a single datagram / TCP message and call
6//! `handle_bytes`. The handler is expected to return a fully-formed
7//! DNS wire response, or `None` if the query was malformed beyond
8//! repair (in which case the transport silently drops — never reply
9//! to a malformed packet, that's an amplification vector).
10//!
11//! Using raw bytes instead of `hickory_proto::op::Message` lets every
12//! transport share the same dispatch without paying double encode/
13//! decode; it also means DoH can pass POST bodies straight through.
14
15use std::net::IpAddr;
16use std::sync::Arc;
17
18use async_trait::async_trait;
19use hickory_proto::op::{Message, MessageType, OpCode, ResponseCode};
20use hickory_proto::serialize::binary::BinDecodable;
21
22#[async_trait]
23pub trait DnsHandler: Send + Sync + 'static {
24    /// Dispatch a single DNS query. `query` is the raw wire format
25    /// (no transport framing). `peer` is the remote IP (the transport
26    /// has already enforced the CIDR allow-list).
27    async fn handle_bytes(&self, query: &[u8], peer: IpAddr) -> Option<Vec<u8>>;
28}
29
30/// Stub handler: parses the query, mirrors the TXID + question section
31/// into a response with RCODE=REFUSED. Replaced by the real recursor
32/// in task #8. Until then imp-dnsd is a well-behaved sink — it answers
33/// (so clients don't retry indefinitely) without leaking anything.
34pub struct RefusedHandler;
35
36#[async_trait]
37impl DnsHandler for RefusedHandler {
38    async fn handle_bytes(&self, query: &[u8], _peer: IpAddr) -> Option<Vec<u8>> {
39        let msg = Message::from_bytes(query).ok()?;
40        let mut resp = Message::new();
41        resp.set_id(msg.id());
42        resp.set_message_type(MessageType::Response);
43        resp.set_op_code(msg.op_code());
44        resp.set_recursion_desired(msg.recursion_desired());
45        resp.set_recursion_available(false);
46        resp.set_response_code(match msg.op_code() {
47            OpCode::Query => ResponseCode::Refused,
48            _ => ResponseCode::NotImp,
49        });
50        for q in msg.queries() {
51            resp.add_query(q.clone());
52        }
53        resp.to_vec().ok()
54    }
55}
56
57// Hickory's Message::to_vec is fallible; wrap the common call sites.
58pub trait MessageVecExt {
59    fn to_vec(&self) -> Result<Vec<u8>, hickory_proto::error::ProtoError>;
60}
61
62impl MessageVecExt for Message {
63    fn to_vec(&self) -> Result<Vec<u8>, hickory_proto::error::ProtoError> {
64        use hickory_proto::serialize::binary::{BinEncodable, BinEncoder};
65        let mut buf = Vec::with_capacity(512);
66        let mut encoder = BinEncoder::new(&mut buf);
67        self.emit(&mut encoder)?;
68        Ok(buf)
69    }
70}

72/// Convenience: make any `Arc<T: DnsHandler>` work as `Arc<dyn DnsHandler>`.
73pub type SharedHandler = Arc<dyn DnsHandler>;

75#[cfg(test)]
76mod tests {
77    use super::*;
78    use hickory_proto::rr::{Name, RecordType};
79    use hickory_proto::op::Query;
80
81    #[tokio::test]
82    async fn refused_stub_mirrors_txid_and_question() {
83        let mut req = Message::new();
84        req.set_id(0x4242);
85        req.set_message_type(MessageType::Query);
86        req.set_op_code(OpCode::Query);
87        req.set_recursion_desired(true);
88        req.add_query(Query::query(
89            Name::from_ascii("example.com.").unwrap(),
90            RecordType::A,
91        ));
92        let bytes = req.to_vec().unwrap();
93
94        let h = RefusedHandler;
95        let resp_bytes = h.handle_bytes(&bytes, "10.0.0.1".parse().unwrap()).await.unwrap();
96        let resp = Message::from_bytes(&resp_bytes).unwrap();
97        assert_eq!(resp.id(), 0x4242);
98        assert_eq!(resp.response_code(), ResponseCode::Refused);
99        assert_eq!(resp.queries().len(), 1);
100        assert!(resp.recursion_desired());
101        assert!(!resp.recursion_available());
102    }
