1//! Per-domain forwarder routing + upstream UDP client.
2//!
3//! Longest-suffix match on the question name picks the forwarder;
4//! servers within a forwarder are tried in the configured order with
5//! a per-attempt timeout. Upstream queries source from a fresh
6//! ephemeral VclDgramSocket so each gets an independent source port
7//! (simple source-port randomisation — enough for a v1 baseline
8//! against Kaminsky-class poisoning).
9//!
10//! TXID randomisation is handled via `rand`; 0x20 case randomisation
11//! is a follow-up (the plumbing is compatible — we'd apply it on the
12//! question name before send + verify on recv).
13
14use std::net::SocketAddr;
15use std::sync::Arc;
16use std::time::Duration;
17
18use anyhow::{anyhow, Context, Result};
19use hickory_proto::op::Message;
20use hickory_proto::rr::Name;
21use hickory_proto::serialize::binary::BinDecodable;
22use rand::Rng;
23use vcl_rs::{VclDgramSocket, VclReactor};
24
25use crate::config::Forwarder as ForwarderCfg;
26
27const DEFAULT_UPSTREAM_TIMEOUT_MS: u64 = 2500;
28const UPSTREAM_BUF: usize = 4096;
29
30#[derive(Clone)]
31pub struct Forwarders {
32    // Sorted longest-first so iter().find() returns the most specific
33    // suffix match in O(n). For imp-sized operator configs (tens of
34    // domains) this is faster than a trie and far simpler.
35    entries: Vec<ForwarderEntry>,
36}
37
38#[derive(Clone, Debug)]
39struct ForwarderEntry {
40    domain: Name,
41    servers: Vec<std::net::IpAddr>,
42}
43
44impl Forwarders {
45    pub fn new(configs: &[ForwarderCfg]) -> Result<Self> {
46        let mut entries = Vec::with_capacity(configs.len());
47        for c in configs {
48            let domain = Name::from_ascii(&c.domain)
49                .with_context(|| format!("bad forwarder domain {:?}", c.domain))?;
50            entries.push(ForwarderEntry {
51                domain: domain.to_lowercase(),
52                servers: c.servers.clone(),
53            });
54        }
55        // Longest-suffix wins → sort by label count descending.
56        entries.sort_by_key(|e| std::cmp::Reverse(e.domain.num_labels()));
57        Ok(Self { entries })
58    }
59
60    /// Return the forwarder whose configured domain is a proper
61    /// suffix of `qname`, preferring the most specific match. A
62    /// domain of "." (the root) would match everything; operators
63    /// who want a global forwarder should configure it explicitly.
64    pub fn lookup(&self, qname: &Name) -> Option<&[std::net::IpAddr]> {
65        let lq = qname.to_lowercase();
66        self.entries
67            .iter()
68            .find(|e| is_suffix(&lq, &e.domain))
69            .map(|e| e.servers.as_slice())
70    }
71}
72
73fn is_suffix(name: &Name, suffix: &Name) -> bool {
74    if name.num_labels() < suffix.num_labels() {
75        return false;
76    }
77    name.zone_of(suffix)
78}

80pub struct UpstreamClient {
81    reactor: VclReactor,
82    timeout: Duration,
83}
84
85impl UpstreamClient {
86    pub fn new(reactor: VclReactor, timeout_ms: Option<u32>) -> Self {
87        let timeout = Duration::from_millis(
88            timeout_ms.map(|t| t as u64).unwrap_or(DEFAULT_UPSTREAM_TIMEOUT_MS),
89        );
90        Self { reactor, timeout }
91    }
92
93    /// Send `query` to the first server that answers (round-robin
94    /// through the list, one timeout per server). `query` is the
95    /// wire-format request from the client; we rewrite the TXID and
96    /// forward, then rewrite the response TXID back on the way out.
97    pub async fn query(
98        &self,
99        servers: &[std::net::IpAddr],
100        query: &[u8],
101    ) -> Result<Vec<u8>> {
102        if servers.is_empty() {
103            return Err(anyhow!("forwarder has no upstream servers"));
104        }
105
106        let mut last_err = None;
107        for server_ip in servers {
108            let peer = SocketAddr::new(*server_ip, 53);
109            match self.query_one(peer, query).await {
110                Ok(resp) => return Ok(resp),
111                Err(e) => {
112                    tracing::debug!(%peer, "upstream query failed: {e}");
113                    last_err = Some(e);
114                }
115            }
116        }
117        Err(last_err.unwrap_or_else(|| anyhow!("no forwarder server responded")))
118    }
119
120    async fn query_one(&self, peer: SocketAddr, orig_query: &[u8]) -> Result<Vec<u8>> {
121        // Bind an ephemeral source socket of the right address family.
122        let sock = if peer.is_ipv4() {
123            VclDgramSocket::bind_ephemeral_v4(self.reactor.clone())
124        } else {
125            VclDgramSocket::bind_ephemeral_v6(self.reactor.clone())
126        }
127        .with_context(|| format!("ephemeral bind for upstream {peer}"))?;
128
129        // Rewrite TXID with a fresh random one for the upstream hop.
130        // We'll restore the client's TXID on the response before
131        // returning to the handler.
132        let orig_msg = Message::from_bytes(orig_query).context("parse client query")?;
133        let client_txid = orig_msg.id();
134
135        let upstream_txid: u16 = rand::thread_rng().gen();
136        let mut out = orig_query.to_vec();
137        out[0..2].copy_from_slice(&upstream_txid.to_be_bytes());
138
139        // Send + await one datagram with a global timeout.
140        sock.send_to(&out, peer)
141            .await
142            .with_context(|| format!("send_to {peer}"))?;
143
144        let mut buf = vec![0u8; UPSTREAM_BUF];
145        let (n, from) = tokio::time::timeout(self.timeout, sock.recv_from(&mut buf))
146            .await
147            .map_err(|_| anyhow!("upstream {peer} timeout"))?
148            .with_context(|| format!("recv_from {peer}"))?;
149
150        // Reject spoofed responses that didn't come from the server
151        // we queried (rudimentary; production 0x20 / cookies add
152        // more layers).
153        if from.ip() != peer.ip() {
154            return Err(anyhow!(
155                "upstream response from unexpected address {from} (wanted {peer})"
156            ));
157        }
158
159        if n < 12 {
160            return Err(anyhow!("short upstream response ({n} bytes)"));
161        }
162        // TXID check
163        let rx_txid = u16::from_be_bytes([buf[0], buf[1]]);
164        if rx_txid != upstream_txid {
165            return Err(anyhow!(
166                "upstream TXID mismatch: got {rx_txid:#06x} expected {upstream_txid:#06x}"
167            ));
168        }
169        buf.truncate(n);
170        // Restore the client's TXID.
171        buf[0..2].copy_from_slice(&client_txid.to_be_bytes());
172        Ok(buf)
173    }
174}

176#[cfg(test)]
177mod tests {
178    use super::*;
179
180    fn fwd(domain: &str, servers: &[&str]) -> ForwarderCfg {
181        ForwarderCfg {
182            domain: domain.into(),
183            servers: servers.iter().map(|s| s.parse().unwrap()).collect(),
184        }
185    }
186
187    #[test]
188    fn longest_suffix_wins() {
189        let configs = vec![
190            fwd("jdt.io", &["10.42.128.19"]),
191            fwd("k8s.jdt.io", &["10.42.113.4"]),
192            fwd("emeraldbroadband.net", &["10.10.90.35", "10.10.90.36"]),
193        ];
194        let f = Forwarders::new(&configs).unwrap();
195
196        let hit = f
197            .lookup(&Name::from_ascii("foo.k8s.jdt.io.").unwrap())
198            .unwrap();
199        assert_eq!(hit, &["10.42.113.4".parse::<std::net::IpAddr>().unwrap()]);
200
201        let hit2 = f.lookup(&Name::from_ascii("www.jdt.io.").unwrap()).unwrap();
202        assert_eq!(
203            hit2,
204            &["10.42.128.19".parse::<std::net::IpAddr>().unwrap()]
205        );
206
207        let hit3 = f
208            .lookup(&Name::from_ascii("ns1.emeraldbroadband.net.").unwrap())
209            .unwrap();
210        assert_eq!(hit3.len(), 2);
211    }
212
213    #[test]
214    fn non_matching_returns_none() {
215        let f = Forwarders::new(&[fwd("jdt.io", &["10.42.128.19"])]).unwrap();
216        assert!(f.lookup(&Name::from_ascii("example.com.").unwrap()).is_none());
217    }
